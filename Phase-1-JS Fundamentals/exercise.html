Q(2)Calculators

let num1=parseInt(prompt("Enter a number:"))
let num2=parseInt(prompt("Enter another number:"))
let operations=prompt("Enter what operation you want to perform(+,-,x,/,%):")
if (operations==='+'){
let result=num1+num2
console.log("The result is:",result)
}else if(operations==='-'){
let result=num1-num2
console.log("The result is:",result)
}else if(operations==='x'){
let result=num1*num2
console.log("The result is:",result)
}else if(operations==='/'){
let result=num1/num2
console.log("The result is:",result)
}else if(operations==='%'){
let result=num1%num2
console.log("The result is:",result)
}else{
console.log("Enter valid operation.")
}

Q(3)Temperature Convertor

let temperature = Number(prompt("Enter temperature:"));
let choice = prompt("Enter C for Celsius to Fahrenheit or F for Fahrenheit to Celsius:");
if (choice === "C") {
let fahrenheit = (temperature * 9 / 5) + 32;
console.log(`${temperature}°C = ${fahrenheit}°F`);
} else if (choice === "F") {
let celsius = (temperature - 32) * 5 / 9;
console.log(`${temperature}°F = ${celsius.toFixed(2)}°C`);
} else {
console.log("Invalid choice");
}
